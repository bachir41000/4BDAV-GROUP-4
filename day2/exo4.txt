1-Écrire un programme permettant de calculer la moyenne des salaires des pilotes dont l’âge est entre 45 et 55 ans

DECLARE
    moyenne_sal pilote.salaire%type;
BEGIN
    SELECT AVG(salaire) INTO moyenne_sal
    FROM pilote
    WHERE age >= 45 and age <= 55;

    DBMS_OUTPUT.PUT_LINE (moyenne_sal || '€');

END;


2-Écrire un programme permettant de calculer le salaire annuel d'un pilote
DECLARE

CREATE OR REPLACE FUNCTION func_annuel_sal (p_matricule pilote.matricule%TYPE)
RETURN NUMBER AS salaire_annuel pilote.salaire%TYPE;
BEGIN
    SELECT salaire * 12 INTO salaire_annuel
FROM pilote
WHERE p_matricule = matricule;
RETURN salaire_annuel;

END;


SELECT
    matricule,
    nom,
    func_annuel_sal(matricule) salaire_ann
